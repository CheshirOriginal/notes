func topKFrequent(nums []int, k int) []int {
    store := make(map[int]int)
    res := make([]int, k)
    
    for _, v := range nums {
        store[v]++
    }
    for i := 0; i < k; i++ {
        max := math.MinInt
        val := 0
        for k, v := range store {
            if v > max {
                max = v
                val = k
            }
        }
        res[i] = val
        delete(store, val)
    }
    return res
}
